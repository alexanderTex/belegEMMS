686  void ScreenPosToWorldRay(
687  int mouseX, int mouseY, // Mouse position, in pixels, from bottom-left corner of the window
688  int screenWidth, int screenHeight, // Window size, in pixels
689  glm::mat4 ViewMatrix, // Camera position and orientation
690  glm::mat4 ProjectionMatrix, // Camera parameters (ratio, field of view, near and far planes)
691  glm::vec3& out_origin, // Ouput : Origin of the ray. /!\ Starts at the near plane, so if you want the ray to start at the camera's position instead, ignore this.
692  glm::vec3& out_direction // Ouput : Direction, in world space, of the ray that goes "through" the mouse.
693  ){
694 
695  // The ray Start and End positions, in Normalized Device Coordinates (Have you read Tutorial 4 ?)
696  glm::vec4 lRayStart_NDC(
697  ((float)mouseX/(float)screenWidth - 0.5f) * 2.0f, // [0,1024] -> [-1,1]
698  ((float)mouseY/(float)screenHeight - 0.5f) * 2.0f, // [0, 768] -> [-1,1]
699  -1.0, // The near plane maps to Z=-1 in Normalized Device Coordinates
700  1.0f
701  );
702  glm::vec4 lRayEnd_NDC(
703  ((float)mouseX/(float)screenWidth - 0.5f) * 2.0f,
704  ((float)mouseY/(float)screenHeight - 0.5f) * 2.0f,
705  0.0,
706  1.0f
707  );
708 
709 
710  // The Projection matrix goes from Camera Space to NDC.
711  // So inverse(ProjectionMatrix) goes from NDC to Camera Space.
712  glm::mat4 InverseProjectionMatrix = glm::inverse(ProjectionMatrix);
713 
714  // The View Matrix goes from World Space to Camera Space.
715  // So inverse(ViewMatrix) goes from Camera Space to World Space.
716  glm::mat4 InverseViewMatrix = glm::inverse(ViewMatrix);
717 
718  glm::vec4 lRayStart_camera = InverseProjectionMatrix * lRayStart_NDC; lRayStart_camera/=lRayStart_camera.w;
719  glm::vec4 lRayStart_world = InverseViewMatrix * lRayStart_camera; lRayStart_world /=lRayStart_world .w;
720  glm::vec4 lRayEnd_camera = InverseProjectionMatrix * lRayEnd_NDC; lRayEnd_camera /=lRayEnd_camera .w;
721  glm::vec4 lRayEnd_world = InverseViewMatrix * lRayEnd_camera; lRayEnd_world /=lRayEnd_world .w;
722 
723 
724  // Faster way (just one inverse)
725  //glm::mat4 M = glm::inverse(ProjectionMatrix * ViewMatrix);
726  //glm::vec4 lRayStart_world = M * lRayStart_NDC; lRayStart_world/=lRayStart_world.w;
727  //glm::vec4 lRayEnd_world = M * lRayEnd_NDC ; lRayEnd_world /=lRayEnd_world.w;
728 
729 
730  glm::vec3 lRayDir_world(lRayEnd_world - lRayStart_world);
731  lRayDir_world = glm::normalize(lRayDir_world);
732 
733 
734  out_origin = glm::vec3(lRayStart_world);
735  out_direction = glm::normalize(lRayDir_world);

739  bool TestRayOBBIntersection(
740  glm::vec3 ray_origin, // Ray origin, in world space
741  glm::vec3 ray_direction, // Ray direction (NOT target position!), in world space. Must be normalize()'d.
742  glm::vec3 aabb_min, // Minimum X,Y,Z coords of the mesh when not transformed at all.
743  glm::vec3 aabb_max, // Maximum X,Y,Z coords. Often aabb_min*-1 if your mesh is centered, but it's not always the case.
744  glm::mat4 ModelMatrix, // Transformation applied to the mesh (which will thus be also applied to its bounding box)
745  float& intersection_distance // Output : distance between ray_origin and the intersection with the OBB
746  ){
747 
748  // Intersection method from Real-Time Rendering and Essential Mathematics for Games
749 
750  float tMin = 0.0f;
751  float tMax = 100000.0f;
752 
753  glm::vec3 OBBposition_worldspace(ModelMatrix[3].x, ModelMatrix[3].y, ModelMatrix[3].z);
754 
755  glm::vec3 delta = OBBposition_worldspace - ray_origin;
756 
757  // Test intersection with the 2 planes perpendicular to the OBB's X axis
758  {
759  glm::vec3 xaxis(ModelMatrix[0].x, ModelMatrix[0].y, ModelMatrix[0].z);
760  float e = glm::dot(xaxis, delta);
761  float f = glm::dot(ray_direction, xaxis);
762 
763  if ( fabs(f) > 0.001f ){ // Standard case
764 
765  float t1 = (e+aabb_min.x)/f; // Intersection with the "left" plane
766  float t2 = (e+aabb_max.x)/f; // Intersection with the "right" plane
767  // t1 and t2 now contain distances betwen ray origin and ray-plane intersections
768 
769  // We want t1 to represent the nearest intersection,
770  // so if it's not the case, invert t1 and t2
771  if (t1>t2){
772  float w=t1;t1=t2;t2=w; // swap t1 and t2
773  }
774 
775  // tMax is the nearest "far" intersection (amongst the X,Y and Z planes pairs)
776  if ( t2 < tMax )
777  tMax = t2;
778  // tMin is the farthest "near" intersection (amongst the X,Y and Z planes pairs)
779  if ( t1 > tMin )
780  tMin = t1;
781 
782  // And here's the trick :
783  // If "far" is closer than "near", then there is NO intersection.
784  // See the images in the tutorials for the visual explanation.
785  if (tMax < tMin )
786  return false;
787 
788  }else{ // Rare case : the ray is almost parallel to the planes, so they don't have any "intersection"
789  if(-e+aabb_min.x > 0.0f || -e+aabb_max.x < 0.0f)
790  return false;
791  }
792  }
793 
794 
795  // Test intersection with the 2 planes perpendicular to the OBB's Y axis
796  // Exactly the same thing than above.
797  {
798  glm::vec3 yaxis(ModelMatrix[1].x, ModelMatrix[1].y, ModelMatrix[1].z);
799  float e = glm::dot(yaxis, delta);
800  float f = glm::dot(ray_direction, yaxis);
801 
802  if ( fabs(f) > 0.001f ){
803 
804  float t1 = (e+aabb_min.y)/f;
805  float t2 = (e+aabb_max.y)/f;
806 
807  if (t1>t2){float w=t1;t1=t2;t2=w;}
808 
809  if ( t2 < tMax )
810  tMax = t2;
811  if ( t1 > tMin )
812  tMin = t1;
813  if (tMin > tMax)
814  return false;
815 
816  }else{
817  if(-e+aabb_min.y > 0.0f || -e+aabb_max.y < 0.0f)
818  return false;
819  }
820  }
821 
822 
823  // Test intersection with the 2 planes perpendicular to the OBB's Z axis
824  // Exactly the same thing than above.
825  {
826  glm::vec3 zaxis(ModelMatrix[2].x, ModelMatrix[2].y, ModelMatrix[2].z);
827  float e = glm::dot(zaxis, delta);
828  float f = glm::dot(ray_direction, zaxis);
829 
830  if ( fabs(f) > 0.001f ){
831 
832  float t1 = (e+aabb_min.z)/f;
833  float t2 = (e+aabb_max.z)/f;
834 
835  if (t1>t2){float w=t1;t1=t2;t2=w;}
836 
837  if ( t2 < tMax )
838  tMax = t2;
839  if ( t1 > tMin )
840  tMin = t1;
841  if (tMin > tMax)
842  return false;
843 
844  }else{
845  if(-e+aabb_min.z > 0.0f || -e+aabb_max.z < 0.0f)
846  return false;
847  }
848  }
849 
850  intersection_distance = tMin;
851  return true;
